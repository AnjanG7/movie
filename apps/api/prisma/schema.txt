model User {
  id           String        @id @default(uuid())
  name         String
  email        String        @unique
  password     String
  status       UserStatus    @default(ACTIVE)
  roleId       String?
  createdAt    DateTime      @default(now())
  updatedAt    DateTime?     @updatedAt
  auditLogs    AuditLog[]
  projectsuser ProjectUser[]
  role         Role?         @relation(fields: [roleId], references: [id])
}

model Role {
  id    String @id @default(uuid())
  name  String @unique
  users User[]
}

model AuditLog {
  id         String   @id @default(uuid())
  entity     String
  action     String
  recordId   String?
  beforeData Json?
  afterData  Json?
  userId     String?
  timestamp  DateTime @default(now())
  user       User?    @relation(fields: [userId], references: [id])

  @@index([entity])
  @@index([recordId])
  @@index([userId])
  @@index([timestamp])
}

model ProjectUser {
